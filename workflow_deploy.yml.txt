name: Build & Deploy (pygbag)

on:
  push:
    branches: [ "main" ]
  workflow_dispatch:

permissions:
  contents: write

jobs:
  build_deploy:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Setup Python
        uses: actions/setup-python@v5
        with:
          python-version: "3.11"

      - name: Install pygbag
        run: |
          python -m pip install --upgrade pip
          python -m pip install pygbag

      - name: Build web (WASM)
        run: |
          pygbag --build src

          if [ -d "src/build/web" ]; then
            rm -rf web && cp -R src/build/web web
          elif [ -d "build/web" ]; then
            rm -rf web && cp -R build/web web
          else
            echo "ERROR: pygbag output not found." >&2
            find . -maxdepth 4 -type d -print
            exit 1
          fi

          touch web/.nojekyll

      - name: Deploy to GitHub Pages (gh-pages)
        uses: JamesIves/github-pages-deploy-action@v4
        with:
          branch: gh-pages
          folder: web
